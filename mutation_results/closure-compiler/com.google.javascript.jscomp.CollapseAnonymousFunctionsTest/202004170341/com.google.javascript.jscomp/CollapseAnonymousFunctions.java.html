/**
24

25 
 * i.e. the following:
26 
 *
27 
 * <pre>
28 
 * var f = function() {}
29 
 * <pre>

 *
31 
 * becomes:
32 
 *
33 
 * <pre>function f() {}</pre>
34

36

37
